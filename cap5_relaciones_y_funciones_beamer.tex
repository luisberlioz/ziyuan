\section{Productos cartesianos y relaciones}
%% ======================================================================

\begin{frame}{Producto cartesiano}
  \begin{block}{Definición}
    Sean $A$ y $B$ conjuntos. El \textbf{producto cartesiano} $A \times B$ es
    el conjunto de todos los \alert{pares ordenados} $(a,b)$ con
    $a \in A$ y $b \in B$:
    \[
      A \times B = \{(a,b) \mid a \in A \text{ y } b \in B\}.
    \]
  \end{block}

  \begin{block}{Tamaño}
    Si $A$ y $B$ son finitos, $|A \times B| = |A| \cdot |B|$.
  \end{block}

  \begin{itemize}
    \item El par es \textbf{ordenado}: $(a,b) \neq (b,a)$ en general.
    \item Se extiende a $n$ conjuntos: $A_1 \times \cdots \times A_n$.
  \end{itemize}
\end{frame}

\begin{frame}{Relaciones}
  \begin{block}{Definición}
    Una \textbf{relación} $\mathcal{R}$ de $A$ en $B$ es cualquier
    subconjunto de $A \times B$.
    Cuando $A = B$, hablamos de una \textbf{relación en} $A$.
  \end{block}

  \begin{block}{Notación}
    En lugar de $(a,b) \in \mathcal{R}$ escribimos $a \mathbin{\mathcal{R}} b$.
  \end{block}

  \begin{exampleblock}{Ejemplos que ya conoce el lector}
    \begin{itemize}
      \item ``$\leq$'' en $\mathbb{Z}$ es una relación en $\mathbb{Z}$.
      \item ``divide a'' en $\mathbb{N}$: $a \mid b$.
      \item ``está matriculado en'' entre alumnos y secciones.
    \end{itemize}
  \end{exampleblock}
\end{frame}

\begin{frame}{Relaciones destacadas: equivalencia y orden}
  Más allá de los ejemplos sueltos, dos \emph{tipos} de relación
  aparecen por todas partes en matemática.

  \begin{block}{Relación de equivalencia}
    Una relación $\mathcal{R}$ en $A$ es de \textbf{equivalencia} si es
    \emph{reflexiva}, \emph{simétrica} y \emph{transitiva}:
    \begin{itemize}
      \item $a \mathbin{\mathcal{R}} a$.
      \item $a \mathbin{\mathcal{R}} b \implies b \mathbin{\mathcal{R}} a$.
      \item $a \mathbin{\mathcal{R}} b,\; b \mathbin{\mathcal{R}} c \implies a \mathbin{\mathcal{R}} c$.
    \end{itemize}
    \textbf{Ejemplo:} $a \equiv b \pmod{n}$ sobre $\mathbb{Z}$. Una relación
    de equivalencia siempre \emph{parte} $A$ en clases disjuntas
    (clases de equivalencia), y recíprocamente toda partición induce una.
  \end{block}

  \begin{block}{Relación de orden (parcial)}
    Es \emph{reflexiva}, \emph{antisimétrica} ($a \mathbin{\mathcal{R}} b$ y
    $b \mathbin{\mathcal{R}} a \implies a = b$) y \emph{transitiva}.
    \textbf{Ejemplos:} $\subseteq$ sobre $\mathcal{P}(X)$, $\leq$ sobre
    $\mathbb{R}$, $\mid$ (divide a) sobre $\mathbb{N}$. Si además todo par
    es comparable, el orden es \textbf{total} (p.\ ej.\ $\leq$ en $\mathbb{R}$).
  \end{block}
\end{frame}

\section{Funciones: definición general}
%% ======================================================================

\begin{frame}{Función: definición}
  \begin{block}{Definición}
    Una \textbf{función} $f$ de $A$ en $B$, escrita $f : A \to B$, es una
    relación de $A$ en $B$ en la que \alert{cada elemento de $A$ aparece
    como primera componente de exactamente un par}.
  \end{block}

  \begin{itemize}
    \item $A$ es el \textbf{dominio} de $f$; $B$ es el \textbf{codominio}.
    \item Si $f(a) = b$: $b$ es la \textbf{imagen} de $a$, y $a$ es una
          \textbf{preimagen} de $b$.
    \item El \textbf{rango} de $f$ es
          $f(A) = \{f(a) \mid a \in A\} \subseteq B$.
  \end{itemize}

  \begin{alertblock}{Idea clave}
    Una función es una \textbf{regla de correspondencia} sin ambigüedad:
    cada $a \in A$ tiene \emph{una y solo una} imagen.
  \end{alertblock}
\end{frame}

\begin{frame}{Funciones como diagramas}
  \begin{center}
    \begin{tikzpicture}[scale=1, every node/.style={font=\small}]
      \node (A) at (0,0) {$A$};
      \node (a1) at (-0.6,-0.7) {$a_1$};
      \node (a2) at (-0.6,-1.3) {$a_2$};
      \node (a3) at (-0.6,-1.9) {$a_3$};
      \node (B) at (4,0) {$B$};
      \node (b1) at (4.6,-0.7) {$b_1$};
      \node (b2) at (4.6,-1.3) {$b_2$};
      \node (b3) at (4.6,-1.9) {$b_3$};
      \draw[->] (a1) -- (b1);
      \draw[->] (a2) -- (b3);
      \draw[->] (a3) -- (b1);
    \end{tikzpicture}
  \end{center}
  \emph{Cada} $a \in A$ tiene exactamente una flecha; algunas $b \in B$ no
  reciben flecha y otras reciben varias.
\end{frame}

\section{Funciones inyectivas y sobreyectivas}
%% ======================================================================

\begin{frame}{Funciones inyectivas}
  \begin{block}{Definición}
    $f : A \to B$ es \textbf{inyectiva} (o \emph{uno a uno}) si elementos
    distintos de $A$ van a elementos distintos de $B$:
    \[
      \forall a_1, a_2 \in A,\; f(a_1) = f(a_2) \implies a_1 = a_2.
    \]
  \end{block}

  \begin{itemize}
    \item Equivale a: $a_1 \neq a_2 \implies f(a_1) \neq f(a_2)$.
    \item Si $A$ es finito y $f$ es inyectiva, entonces $|A| \leq |B|$.
  \end{itemize}

  \begin{exampleblock}{Idea visual}
    Las flechas de la función \textbf{nunca se juntan} en el codominio.
  \end{exampleblock}
\end{frame}

\begin{frame}{Funciones sobreyectivas}
  \begin{block}{Definición}
    $f : A \to B$ es \textbf{sobreyectiva} (o \emph{subyectiva}) si todo
    elemento de $B$ es imagen de algún elemento de $A$:
    \[
      \forall b \in B,\; \exists a \in A \text{ con } f(a) = b.
    \]
    Es decir, $f(A) = B$.
  \end{block}

  \begin{itemize}
    \item Si $A$ y $B$ son finitos y $f$ es sobreyectiva, entonces $|A| \geq |B|$.
    \item \textbf{Biyectiva} = inyectiva \emph{y} sobreyectiva.
  \end{itemize}

  \begin{alertblock}{Idea visual}
    Las flechas \textbf{cubren todo} el codominio: cada $b \in B$ recibe al menos una.
  \end{alertblock}
\end{frame}

\section{Conteo de funciones sobreyectivas: Stirling}
%% ======================================================================

\begin{frame}{Números de Stirling del segundo tipo}
  \begin{block}{Definición}
    Para enteros $n, k \geq 0$, el \textbf{número de Stirling de segundo
    tipo} $S(n,k)$ es el \alert{número de formas de particionar un conjunto
    de $n$ elementos en $k$ subconjuntos no vacíos}.
  \end{block}

  \begin{itemize}
    \item Casos base: $S(n,0) = 0$ si $n > 0$, $S(0,0) = 1$, $S(n,1) = S(n,n) = 1$.
    \item \textbf{Recurrencia}:
      \[
        S(n,k) = k \cdot S(n-1,k) + S(n-1,k-1).
      \]
      El elemento nuevo se une a uno de los $k$ bloques existentes
      (\,$\,k\,S(n-1,k)\,$) \emph{o} forma un bloque aparte
      (\,$S(n-1,k-1)\,$).
  \end{itemize}
\end{frame}

\begin{frame}{Funciones sobreyectivas por conteo}
  \begin{block}{Teorema}
    El número de funciones sobreyectivas $f : A \to B$ con $|A| = n$,
    $|B| = k$ y $n \geq k$ es
    \[
      k! \cdot S(n,k).
    \]
  \end{block}
  \textbf{Idea:} particionar $A$ en $k$ bloques no vacíos
  ($\,S(n,k)$ formas) y asignar cada bloque a un elemento de $B$
  ($\,k!$ formas).
\end{frame}

\section{Funciones especiales}
%% ======================================================================

\begin{frame}{Funciones constantes, identidad, piso y techo}
  \begin{block}{Función constante}
    $f : A \to B$ es \textbf{constante} si existe $b_0 \in B$ tal que
    $f(a) = b_0$ para todo $a \in A$.
  \end{block}

  \begin{block}{Función identidad}
    La \textbf{identidad} en $A$ es $\mathrm{id}_A : A \to A$ definida por
    $\mathrm{id}_A(a) = a$. Es \emph{biyectiva} y actúa como neutro para la composición.
  \end{block}

  \begin{block}{Función piso y techo}
    Para $x \in \mathbb{R}$:
    \[
      \lfloor x \rfloor = \text{mayor entero } \leq x, \qquad
      \lceil x \rceil = \text{menor entero } \geq x.
    \]
  \end{block}
\end{frame}

\begin{frame}{Función característica y factorial}
  \begin{block}{Función característica}
    Para $X \subseteq U$,
    \[
      \chi_X : U \to \{0,1\}, \quad
      \chi_X(u) = \begin{cases} 1 & \text{si } u \in X, \\ 0 & \text{si } u \notin X. \end{cases}
    \]
    Codifica subconjuntos como funciones: $\mathcal{P}(U) \leftrightarrow \{0,1\}^U$.
  \end{block}

  \begin{block}{Factorial y combinaciones}
    Para $n \geq 0$: $n! = n \cdot (n-1) \cdots 1$ (con $0! = 1$).
    Para $0 \leq r \leq n$:
    \[
      \binom{n}{r} = \frac{n!}{r!\,(n-r)!}
    \]
    cuenta los subconjuntos de tamaño $r$ de un conjunto de $n$ elementos.
  \end{block}
\end{frame}

\section{El principio del palomar}
%% ======================================================================

\begin{frame}{El principio}
  \begin{block}{Principio del palomar}
    Si $n$ objetos se distribuyen en $k$ cajas y $n > k$, entonces
    \alert{al menos una caja contiene más de un objeto}.
  \end{block}

  \begin{block}{Versión general}
    Si $n$ objetos van a $k$ cajas, alguna caja contiene al menos
    $\lceil n/k \rceil$ objetos.
  \end{block}

  \begin{alertblock}{Carácter no constructivo}
    El principio \emph{garantiza la existencia} del conflicto, pero
    \emph{no dice} dónde ocurre. Es un argumento de conteo disfrazado.
  \end{alertblock}
\end{frame}

\begin{frame}{Ejemplos que conectan con el resto del capítulo}
  \begin{exampleblock}{Ejemplo 1 — divisibilidad}
    Entre $\{1,2,\dots,n+1\}$ hay dos que dejan el mismo residuo al dividir
    por $n$: hay $n+1$ números y solo $n$ residuos posibles.
  \end{exampleblock}

  \begin{exampleblock}{Ejemplo 2 — funciones no inyectivas}
    Toda función $f : \{1,\dots,n+1\} \to \{1,\dots,n\}$ \emph{no} es inyectiva.
    Es exactamente la versión ``funciones'' del principio del palomar.
  \end{exampleblock}

  \begin{exampleblock}{Ejemplo 3 — cumpleaños}
    Con 367 personas hay dos que cumplen años el mismo día;
    con sólo 23, la probabilidad ya supera el 50\%.
  \end{exampleblock}
\end{frame}

\section{Composición de funciones e inversas}
%% ======================================================================

\begin{frame}{Composición: encadenar funciones}
  \begin{block}{Definición}
    Dadas $g : A \to B$ y $f : B \to C$, la \textbf{composición} $f \circ g :
    A \to C$ se define por
    \[
      (f \circ g)(a) = f(g(a)), \quad a \in A.
    \]
  \end{block}

  \begin{itemize}
    \item \textbf{No es conmutativa} en general: $f \circ g \neq g \circ f$.
    \item \textbf{Identidad}: $f \circ \mathrm{id}_A = f = \mathrm{id}_B \circ f$.
    \item \textbf{Asociatividad}: $(f \circ g) \circ h = f \circ (g \circ h)$.
    \item Componer inyectivas (resp.\ sobreyectivas) da inyectiva (resp.\ sobreyectiva).
  \end{itemize}
\end{frame}

\begin{frame}{Función inversa: deshacer una función}
  \begin{block}{Definición}
    Si $f : A \to B$ es \textbf{biyectiva}, su \textbf{inversa}
    $f^{-1} : B \to A$ se define por
    \[
      f^{-1}(b) = a \iff f(a) = b.
    \]
  \end{block}

  \begin{itemize}
    \item $f \circ f^{-1} = \mathrm{id}_B$ y $f^{-1} \circ f = \mathrm{id}_A$.
    \item $f$ tiene inversa $\iff$ $f$ es biyectiva.
    \item Si $f$ y $g$ son biyectivas, $(g \circ f)^{-1} = f^{-1} \circ g^{-1}$
          (el orden se \emph{invierte}).
  \end{itemize}

  \begin{alertblock}{Idea}
    Las biyecciones son las funciones \textbf{reversibles}. Por eso la
    biyectividad es tan importante en matemática y en criptografía.
  \end{alertblock}
\end{frame}

\section{Complejidad computacional}
%% ======================================================================

\begin{frame}{¿Qué pregunta la complejidad computacional?}
  \begin{block}{Idea}
    La \textbf{complejidad computacional} clasifica los problemas según
    los \emph{recursos} (tiempo, memoria) que necesita cualquier algoritmo
    que los resuelva, en función del tamaño $n$ de la entrada.
  \end{block}

  \begin{block}{Vocabulario esencial}
    \begin{itemize}
      \item \textbf{Problema}: pregunta genérica a resolver.
      \item \textbf{Instancia}: un caso particular con datos concretos.
      \item \textbf{Algoritmo}: procedimiento paso a paso que resuelve
            el problema.
      \item \textbf{Recurso}: tiempo (número de operaciones) o espacio
            (memoria).
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{Notación asintótica}
  \begin{block}{Definición}
    Decimos $f(n) = O(g(n))$ si existen constantes $C, n_0 > 0$ tales que
    \[
      |f(n)| \leq C \cdot g(n) \quad \text{para todo } n \geq n_0.
    \]
  \end{block}

  \begin{itemize}
    \item $O(\cdot)$ describe el \textbf{comportamiento asintótico}: ignora
          constantes y términos de menor orden.
    \item Útil porque dos algoritmos con constantes distintas pero misma
          $O(\cdot)$ escalan igual cuando $n$ crece.
  \end{itemize}
\end{frame}

\begin{frame}{Jerarquía de crecimiento}
  \begin{block}{Orden, de menor a mayor}
    \[
      1 \;<\; \log n \;<\; n \;<\; n \log n \;<\; n^2 \;<\; n^3 \;<\; 2^n.
    \]
  \end{block}

  \begin{itemize}
    \item \textbf{Tiempo polinomial}: $O(n^k)$ para alguna $k$
          (clase \textbf{P}).
    \item \textbf{Tiempo exponencial}: $O(c^n)$ con $c > 1$.
    \item La frontera polinomial/exponencial \emph{no} es sólo un tecnicismo:
          marca la diferencia entre \textbf{resoluble} y
          \textbf{intratable} en la práctica.
  \end{itemize}
\end{frame}

\section{Análisis de algoritmos}
%% ======================================================================

\begin{frame}{Análisis de algoritmos: precisar la intuición}
  \begin{block}{Definición}
    El \textbf{análisis de algoritmos} determina, asintóticamente, el número
    de operaciones elementales que un algoritmo realiza, en función del
    tamaño $n$ de la entrada.
  \end{block}

  \begin{block}{Tres medidas estándar}
    \begin{itemize}
      \item \textbf{Peor caso}: $T_{\max}(n) = \max_{|x|=n} (\text{ops}(x))$.
            Cota superior universal.
      \item \textbf{Caso promedio}: $T_{\text{avg}}(n) =
            \mathbb{E}[\text{ops}(X)]$ sobre una distribución de entradas.
      \item \textbf{Mejor caso}: $T_{\min}(n) = \min_{|x|=n} (\text{ops}(x))$.
            Útil sólo como información complementaria.
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{Ejemplo: búsqueda lineal}
  \begin{exampleblock}{Problema}
    Buscar un elemento $x$ en un arreglo $A[1..n]$ recorriendo las
    posiciones en orden y devolviendo ``no está'' si no aparece.
  \end{exampleblock}

  \begin{itemize}
    \item \textbf{Mejor caso}: $x = A[1]$, $T_{\min}(n) = O(1)$.
    \item \textbf{Peor caso}: $x \notin A$ o $x = A[n]$,
          $T_{\max}(n) = O(n)$.
    \item \textbf{Caso promedio}: $x$ en posición uniforme,
          $T_{\text{avg}}(n) = O(n)$.
  \end{itemize}

  \begin{alertblock}{Conclusión}
    La búsqueda lineal es $O(n)$ en el peor caso.
    Un análisis cuidadoso \emph{distingue} mejor / peor / promedio, no los
    mezcla.
  \end{alertblock}
\end{frame}
